### Ответ

```js
function objSum(arr) {

  return arr.reduce((acc, el) => {
    acc[el] = (acc[el] | 0) + 1 
    return acc
  }, {})
}

console.log(objSum(arr)); // {'1:1, 2:1, 3:1, 4:2'}
```
