const funcFunc = (obj) => {
  let sum = 0;
  const stack = [obj];

  while (stack.length > 0) {
    const currentNode = stack.pop();
    sum += currentNode.val;

    if (currentNode.left) {
      stack.push(currentNode.left);
    }

    if (currentNode.right) {
      stack.push(currentNode.right);
    }
  }

  return sum;
};
