### Ответ

```js
function flattenObj(arr) {
  let stack = [];

  for (let i = 0; i < arr.length; i++) {
    if (typeof arr[i] === 'object') {
      stack.push({ id: arr[i].id });
      if (Array.isArray(arr[i].next)) {
        stack = stack.concat(flattenObj(arr[i].next));
      }
    }
  }

  return stack;
}
```
