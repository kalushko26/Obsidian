  arr.forEach((item, index) => {
    setTimeout(() => {
      console.log(item.name);
    }, item.timeout);
  });
}

const arr = [
  {name:'first', timeout:3000},
  {name:'second', timeout:5000},
  {name:'third', timeout:2000}
