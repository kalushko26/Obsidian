```js
function anagram(str) {
  let ang = {}
  
  for (let word of str) {
    const sorted = word.split('').sort().join('')
    if(ang[sorted]){
      ang[sorted].push(word)
    }else{
      ang[sorted] = [word]
    }
  }
  return Object.values(ang).filter(el => el.length > 1)
}
```
