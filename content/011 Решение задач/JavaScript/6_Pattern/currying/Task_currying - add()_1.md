```js
let add = function (a = 0) {
 // Ваш код здесь
}

console.log(add(1)(2)(3)()); // 6
console.log(add(1)(2)()); // 3
```

### Ответ

```js
function add(a = 0) {
	let sum = a;
	const res = (b) => {
		if(typeof b === 'undefined') return sum
		sum += b;
		return res
	}
	return res
}

console.log(add(1)(2)(3)()) // 6
console.log(add(1)(2)()) // 3
```
